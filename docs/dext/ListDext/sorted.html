List<T> sorted({
  Comparable Function(T t)? comparator,
  bool descending = false,
}) {
  List<T> resultList;
  if (comparator == null) {
    resultList = toList()
      ..sort((a, b) {
        if (a == null) {
          return -1;
        }
        if (b == null) {
          return 1;
        }

        if (a is Comparable && b is Comparable) {
          return a.compareTo(b);
        }

        return 0;
      });
  } else {
    resultList = toList()
      ..sort((a, b) => comparator(a).compareTo(comparator(b)));
  }

  if (descending) {
    resultList = resultList.reversed.toList();
  }

  return resultList;
}
